Blk  : Code Blk|If Blk|While Blk|For Blk| ;
Code : Eval|Out|Exp|Var_dec;
Eval : IDENTIFIER ASSIGN Exp STMT_TERMINATOR;

Exp : Val op Exp| OPEN_PARANTHESIS Exp CLOSE_PARANTHESIS | Val;
Val : IDENTIFIER | STRING | DECIMAL ;
op : ARITH | BITWISE ;
Out : KW_PRINTLN OPEN_PARANTHESIS Body CLOSE_PARANTHESIS STMT_TERMINATOR;
Body : STRING | STRING COMMA Val|;

Var_dec : LET IDENTIFIER COLON DATATYPE ASSIGN Val STMT_TERMINATOR {printf("trying to add\n");add_value($2,$6);printf("added\n");}
        ;

If      : IF Exp OPEN_BLOCK Blk CLOSE_BLOCK Else ; 
Else    : ELSE OPEN_BLOCK Blk CLOSE_BLOCK | ;

While   : WHILE Exp OPEN_BLOCK Blk CLOSE_BLOCK;

For : FOR IDENTIFIER IN T_const DOT DOT T_const OPEN_BLOCK Blk CLOSE_BLOCK;
